            // Grammar check: if grammar permits trailing comma, only the .comma branch (double-comma) emits;
            // if grammar disallows trailing comma, both branches emit. CHOOSE based on the grammar read above.
            if (parser.peek().tag == .comma /* OR: include `or parser.peek().tag == .r_paren` if grammar disallows trailing */) {
                try parser.diagnostics.emit(
                    diagnostics_mod.Codes.e_empty_arg_comma,
                    .err,
                    "expected argument between commas",
                    comma_tok.span,
                );
                continue; // skip the bad comma; loop will re-check
            }
            // Trailing comma after a valid arg: if grammar permits, accept; if disallows, emit.
        }
        const arg = try parseExpression(parser, 0);
        try args.append(allocator, arg);
    }

    return try args.toOwnedSlice(allocator);
}
