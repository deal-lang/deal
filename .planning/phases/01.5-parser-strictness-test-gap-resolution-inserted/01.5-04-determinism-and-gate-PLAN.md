Purpose: D-18 promises alphabetical JSON key emission for byte-stable snapshots, but the only existing guard is the snapshot byte-diff test — which proves the FIRST parse matches the locked snapshot, NOT that the SECOND parse matches the first. A non-deterministic emitter (e.g. accidental `std.HashMap` iteration order, accidental wall-clock timestamps, accidental non-canonical float formatting) would only surface after the locked snapshot drifts naturally. The determinism test catches that class of regression at the contract surface — same source, same process, two parses, byte-identical output.

**FORBIDDEN in this test:** `deal_parse_internal` and `deal_free_internal`. Those are test-only helpers; the determinism contract must hold at the EXTERN C ABI surface external callers actually use.

The C ABI lifetime contract (from lib.zig:284): the JSON buffer returned by `deal_ast_json` is owned by the handle's arena. `deal_free(handle)` runs `arena.deinit()`, which invalidates the returned pointer. Therefore: copy parse-A's JSON into a gpa-owned buffer BEFORE calling `deal_free(handle_a)` AND BEFORE starting parse B (so the two byte strings live simultaneously for comparison).

Determinism test concrete shape (mirrors CONTEXT.md §specifics — the C ABI version):
```zig
test "determinism.parse_twice" {
    const gpa = std.testing.allocator;
    const io = std.testing.io;
    const cwd = std.Io.Dir.cwd();

    for (showcase_files) |path| {
        const source = try cwd.readFileAlloc(io, path, gpa, .unlimited);
        defer gpa.free(source);

        // First parse — capture AST JSON via the PUBLIC C ABI.
        const handle_a = lib.deal_parse(source.ptr, source.len, path.ptr, path.len);
        // Defer-free A AFTER we've copied its JSON below (so we don't accidentally free
        // before the dupe completes). The cleanest pattern is an explicit deal_free at end.
        var ptr_a: [*]const u8 = undefined;
        var len_a: usize = 0;
        const ok_a = lib.deal_ast_json(handle_a, &ptr_a, &len_a);
        try std.testing.expect(ok_a);
        // Copy bytes into a gpa-owned buffer so they survive deal_free(handle_a).
        const json_a = try gpa.dupe(u8, ptr_a[0..len_a]);
        defer gpa.free(json_a);

        // Same for diagnostics JSON (claude's-discretion addon — trivially cheap).
        var dptr_a: [*]const u8 = undefined;
        var dlen_a: usize = 0;
        const dok_a = lib.deal_diagnostics_json(handle_a, &dptr_a, &dlen_a);
        try std.testing.expect(dok_a);
        const diag_a = try gpa.dupe(u8, dptr_a[0..dlen_a]);
        defer gpa.free(diag_a);

        // Free handle A — invalidates ptr_a / dptr_a, but json_a / diag_a still live.
        lib.deal_free(handle_a);

        // Second parse of the same source — must produce byte-identical JSON.
        const handle_b = lib.deal_parse(source.ptr, source.len, path.ptr, path.len);
        defer lib.deal_free(handle_b);

        var ptr_b: [*]const u8 = undefined;
        var len_b: usize = 0;
        const ok_b = lib.deal_ast_json(handle_b, &ptr_b, &len_b);
        try std.testing.expect(ok_b);

        var dptr_b: [*]const u8 = undefined;
        var dlen_b: usize = 0;
        const dok_b = lib.deal_diagnostics_json(handle_b, &dptr_b, &dlen_b);
        try std.testing.expect(dok_b);

        // Byte-equality across the two parses.
        if (!std.mem.eql(u8, json_a, ptr_b[0..len_b])) {
            std.debug.print("determinism.parse_twice DIVERGED on AST JSON for {s}\n", .{path});
            try std.testing.expectEqualStrings(json_a, ptr_b[0..len_b]); // will fail with diff
        }
        if (!std.mem.eql(u8, diag_a, dptr_b[0..dlen_b])) {
            std.debug.print("determinism.parse_twice DIVERGED on diagnostics JSON for {s}\n", .{path});
            try std.testing.expectEqualStrings(diag_a, dptr_b[0..dlen_b]);
        }
    }
}
```
