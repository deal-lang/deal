```zig
fn parseCalcBody(p: *Parser) !*ast.Node {
    const open = try p.expect(.l_brace);
    var bindings: std.ArrayList(*ast.Node) = .empty;
    var annots: std.ArrayList(*ast.Node) = .empty;
    var return_stmt: ?*ast.Node = null;

    while (p.peek().tag != .r_brace and p.peek().tag != .eof) {
        switch (p.peek().tag) {
            .kw_return => {
                return_stmt = try parseReturnStatement(p);
                // ReturnStatement must be last â€” break after consuming it.
                break;
            },
            .kw_derived, .kw_attribute => {
                // LocalBinding â€” reuse existing parseAttributeUsage path
                const binding = try parseMemberDeclaration(p);
                if (binding) |b| try bindings.append(p.arena, b);
            },
            .annotation, .annotation_prefix => {
                const ann = try parseAnnotationStatement(p);
                try annots.append(p.arena, ann);
            },
            else => {
                // Unexpected â€” emit E0100 and sync (mirrors parseDefinitionBody else arm)
                try p.diags.append(p.arena, .{
                    .code = "E0100",
                    .severity = .err,
                    .message = "unexpected token in calc body",
                    .span = p.peek().span,
                });
                syncToStatement(p);
            },
        }
    }
    // E2601 if no ReturnStatement found:
    if (return_stmt == null) {
        try p.diags.append(p.arena, .{
            .code = "E2601",
            .severity = .err,
            .message = "calc body must end with a `return` statement",
            .span = open.span,
        });
        // Synthesize a dummy return_stmt for error recovery.
    }
    const close = try p.expect(.r_brace);
    const span = ast.Span{ .start = open.span.start, .end = close.span.end };
    return p.makeNode(.calc_body, span, .{
        .calc_body = .{
            .bindings = try bindings.toOwnedSlice(p.arena),
            .return_stmt = return_stmt orelse ..., // error-recovery node
            .annotations = try annots.toOwnedSlice(p.arena),
        },
    });
}
```

```zig
    const open = try p.expect(.l_brace);

    var annots: std.ArrayList(*ast.Node) = .empty;

    while (p.peek().tag != .r_brace and p.peek().tag != .eof) {
```

```zig
fn parseReturnContract(p: *Parser) !*ast.Node {
    const arrow_tok = p.advance(); // consume .arrow ("=>")
    var items: std.ArrayList(*ast.Node) = .empty;

    while (true) {
        const tok = p.peek();
        const item: *ast.Node = blk: {
            if (tok.tag == .ident and std.mem.eql(u8, p.tokenText(tok.span), "sig")) {
                // Contextual: "sig" as IDENT recognized ONLY here after "=>"
                break :blk try parsePrecisionSpecSig(p);
            } else if (tok.tag == .plus_minus) {
                break :blk try parsePrecisionSpecPlusMinus(p);
            } else if (tok.tag == .plus and p.peekAhead(1).tag == .slash
                       and p.peekAhead(2).tag == .minus) {
                // +/- ASCII alias â€” consume 3 tokens, parse as tolerance
                break :blk try parsePrecisionSpecPlusSlashMinus(p);
            } else if (tok.tag == .ident) {
                // ConstraintRef: QualifiedName (may include . separators) + optional args
                break :blk try parseConstraintRef(p);
            } else {
                break; // no more ContractItems; caller sees "{" or eof
            }
        };
        try items.append(p.arena, item);
        if (p.peek().tag == .comma) {
            _ = p.advance(); // consume ","
        } else {
            break;
        }
    }
    const span = ast.Span{ .start = arrow_tok.span.start, .end = p.peek().span.start };
    return p.makeNode(.return_contract, span, .{
        .return_contract = .{ .items = try items.toOwnedSlice(p.arena) },
    });
}
```

```zig
fn checkConstraintRefCycle(
    a: *Analyzer,
    current_name: []const u8,
    target_name: []const u8,
    span: ast.Span,
) !void {
    // Mirror checkSpecializationCycle exactly:
    // 1. a.constraint_chain_map: StringHashMap([]const u8) â€” maps constraint name â†’ its first ConstraintRef name
    // 2. visited: StringHashMap(void), seed with current_name
    // 3. Walk cursor = target_name, following constraint_chain_map
    // 4. If cursor in visited â†’ emit E2611 (cycle variant) instead of E2300
    const max_hops = a.table.entries.count() + 1;
    var seen = std.StringHashMap(void).init(a.arena);
    try seen.put(current_name, {});
    var cursor = lastSegment(target_name);
    var hops: usize = 0;
    while (hops < max_hops) : (hops += 1) {
        if (seen.contains(cursor)) {
            try a.collector.emitFmt(
                Codes.e_constraint_ref_not_found,
                .err, span,
                "ConstraintRef cycle detected involving `{s}`", .{cursor},
            );
            return;
        }
        try seen.put(cursor, {});
        const next = a.constraint_chain_map.get(cursor) orelse return;
        cursor = lastSegment(next);
    }
}
```
